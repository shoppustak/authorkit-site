    Create postcss.config.js with optimized CSS pipeline and update build scripts:

    1. Create postcss.config.js with this exact configuration:
    ```javascript
    module.exports = {
      plugins: [
        require('tailwindcss'),
        ...(process.env.NODE_ENV === 'production' ? [
          require('@fullhuman/postcss-purgecss')({
            content: [
              './*.html',
              './includes/**/*.html',
              './js/**/*.js'
            ],
            defaultExtractor: content => content.match(/[\w-/:]+(?<!:)/g) || [],
            safelist: {
              standard: [/^bg-/, /^text-/, /^border-/, /^hover:/, /^focus:/],
              deep: [],
              greedy: []
            }
          }),
          require('cssnano')({
            preset: ['default', {
              discardComments: { removeAll: true },
              normalizeWhitespace: true,
              minifyFontValues: { removeQuotes: false },
              colormin: false // Preserve black/white values
            }]
          })
        ] : [])
      ]
    }
    ```

    2. Update package.json scripts to consolidate CSS files:
    ```json
    "build:css:prod": "NODE_ENV=production postcss css/input.css -o css/styles.min.css",
    "build:css:merge": "cat css/design-tokens.css css/monochrome-components.css css/style.css > css/merged-legacy.css",
    "build:css:full": "npm run build:css:merge && NODE_ENV=production postcss css/input.css -o css/styles.min.css --config postcss.config.js",
    "build:prod": "npm run build:css:full && npm run build:js",
    "test:css-size": "ls -lh css/styles.min.css | awk '{print $5}'"
    ```

    This pipeline will:
    - Use PurgeCSS to remove unused Tailwind classes
    - Use cssnano to minify and optimize the CSS
